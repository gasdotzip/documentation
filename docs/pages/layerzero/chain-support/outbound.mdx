import lzConfigData from '../../../../data/layerzero/lzConfigData.json'

export const generateTableRows = (config) => {
const excludedChains = ['ShimmerEVM', 'Canto', 'DOS']
// 30198 = Beam
// 30176 = Meter
const unsupportedV2Ids = [30176]

const filteredConfig = Object.entries(config).filter(
([, value]) => !excludedChains.includes(value.name)
)

return filteredConfig.map(([key, value]) => {
const displayIdV2 = unsupportedV2Ids.includes(value.idv2) ? 'Not Supported' : value.idv2
return (

<tr key={key} className="vocs_TableRow">
  <td className="vocs_TableCell">{value.name || 'Not Supported'}</td>
  <td className="vocs_TableCell">{value.id || 'Not Supported'}</td>
  <td className="vocs_TableCell">{displayIdV2}</td>
</tr>
) }) }

export const generateTableHeader = () => {
  return (
    <tr>
      <th className="vocs_TableHeader">Chain Name</th>
      <th className="vocs_TableHeader">LayerZero ID v1</th>
      <th className="vocs_TableHeader">LayerZero ID v2</th>
    </tr>
  )
}
